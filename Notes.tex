		\section{Linear Algebra}
		\subsection{Linear Independence}
		\begin{itemize}
			\item consider $\boldsymbol{X}$: $m\times n$ (a vector of vectors) and $\boldsymbol{a}$: $n\times 1$ (a vector of constants)
			\item $\boldsymbol{X}=[\boldsymbol{x}_{1}~\boldsymbol{x}_{2}~...~\boldsymbol{x}_{n}]$
			\item $\boldsymbol{Xa}=\sum\limits^{n}_{i=1}a_{i}\boldsymbol{x}_{i}=\boldsymbol{0}$
			\item $\boldsymbol{a}=\boldsymbol{0}$ is the trivial solution
			\item if $\boldsymbol{a}=\boldsymbol{0}$ is the only solution, then the vectors $\boldsymbol{x}_{i}$ are \textbf{linearly independent}
			\item if at least one $a_{i}\not=0$, then the vectors $\boldsymbol{x}_{i}$ are \textbf{linearly dependent}
			\item rearranging $\sum\limits^{n}_{i=1}a_{i}\boldsymbol{x}_{i}=0$:
			\begin{equation*}\begin{split}
				\boldsymbol{x}_{i}=\frac{a_{i}}{a_{i}}\boldsymbol{x}_{i}+...+a_{i}\boldsymbol{x}_{i}
			\end{split}\end{equation*}
		\end{itemize}
